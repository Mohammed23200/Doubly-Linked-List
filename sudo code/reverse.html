Reverse

  • FUNCTION Reverse()
  • SET current = head
  • SET prev = NULL
  • WHILE current is NOT NULL DO
  • SET nextNode = current.Next
  • SET current TO current.Next
  • SET current.Prev = nextNode
  • SET prev = current
  • SET current = nextNode
  • END WHILE
  • SET head = prev
  • END FUNCTION
