  • PRINT "List doesn't have enough elements!"
  • RETURN
  • END IF
  • SET max1 = MINIMUM_INTEGER_VALUE
  • SET max2 = MINIMUM_INTEGER_VALUE
  • SET current = head
  • WHILE current is NOT NULL DO
  • IF current.value > max1 THEN
  • SET max2 = max1

  • ELSE IF current.value > max2 AND current.value is NOT equal to max1 THEN
  • SET max2 = current.value
  • END IF

  • END WHILE
