  • FUNCTION Insert(k, x)
  • IF k < 1 THEN
  • THROW "Invalid position"
  • ENDIF
  • newNode = CREATE new Node with value x
  • IF k == 1 THEN
  • newNode.Next = head
  • IF head IS NOT NULL THEN
  • head.Prev = newNode
  • ENDIF
  • head = newNode
  • ELSE
  • current = head
  • index = 1
  • WHILE current IS NOT NULL AND index < k - 1 DO
  • current = current.Next
  • index = index + 1
  • END WHILE
  • IF current IS NULL THEN
  • THROW "Invalid position"
  • ENDIF
  • newNode.Next = current.Next
  • IF current.Next IS NOT NULL THEN
  • current.Next.Prev = newNode
  • ENDIF
  • current.Next = newNode
  • newNode.Prev = current
  • ENDIF
  • RETURN this
  • END FUNCTION
